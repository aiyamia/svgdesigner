  //console.log(`删除--当前this：\n${Object.keys(currentGroup.children)}`);
}else{
  if(this_target.type_id != currentGroup.type_id){
    currentGroup.addChild(this_target)
    // console.log(`添加--当前this：\n${Object.keys(currentGroup.children)}`);
  }
}
